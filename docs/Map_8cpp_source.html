100  this->_size.y++;
101  this->_size.x = std::max(this->_size.x, line.size());
102  for (auto it = lines.begin(); it < lines.end(); it++)
103  it->resize(this->_size.x, ' ');
104  lines.push_back(line);
105  }
106 
107  for (unsigned y = 0; y < lines.size(); y++) {
108  for (unsigned x = 0; x < lines[y].size(); x++)
109  try {
110  auto c = lines[y][x];
111 
112  if (c == '0') {
113  this->_startPoints.emplace_back(x, y);
114  c = ' ';
115  }
116  auto &p = this->_objects.emplace_back(Map::_blockBuilders.at(c)(this->_game));
117  auto pos = p->getPosition();
118 
119  pos.x = x * 64;
120  pos.y = y * 64;
121  p->setPosition(pos);
122  } catch (std::exception &) {
123  throw InvalidSavedMapException(std::string("No object can be built for character '") + lines[y][x] + "' (ASCII " + std::to_string(lines[y][x]) + ")");
124  }
125  }
