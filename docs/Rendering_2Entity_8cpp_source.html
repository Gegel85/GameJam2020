19  void Entity::Config::loadFile(Resources &resources, const std::string &path)
20  {
21  std::ifstream stream{path};
22  nlohmann::json value;
23 
24  if (stream.fail())
25  throw InvalidAnimationConfigException("Cannot open file " + path);
26 
27  try {
28  unsigned counter = 0;
29 
30  stream >> value;
31  this->texture = value["texture"];
32  this->tileSize = {value["tileSize"]["x"], value["tileSize"]["y"]};
33  for (auto &val : value["animations"]) {
34  if (val == 0)
35  throw InvalidAnimationConfigException("Animation count can't be 0");
36  this->animations.push_back(val);
37  }
38  for (auto &val : value["delays"])
39  this->delays.push_back(val);
40  for (int i = 0; i < DEAD; i++)
41  for (int j = 0; j < NB_DIRS; j++) {
42  this->animationStart.emplace_back(counter);
43  counter += this->animations[i];
44  }
45  this->animationStart.emplace_back(counter);
46  this->textureSize = resources.textures.at(this->texture).getSize();
47  if (this->textureSize.x % this->tileSize.x || this->textureSize.y % this->tileSize.y)
48  throw InvalidAnimationConfigException(
49  "The texture chosen (" + this->texture +
50  ") size isn't a perfect grid of " +
51  std::to_string(this->tileSize.x) + "x" + std::to_string(this->tileSize.y) +
52  " tiles"
53  );
54  } catch (std::out_of_range &) {
55  throw InvalidAnimationConfigException("The texture given is not loaded: " + this->texture);
56  } catch (nlohmann::detail::parse_error &e) {
57  throw InvalidAnimationConfigException("The JSON file has an invalid format: " + std::string(e.what()));
58  } catch (nlohmann::detail::type_error &e) {
59  throw InvalidAnimationConfigException("The JSON values are invalid: " + std::string(e.what()));
60  }
61  if (this->animations.size() != DEAD + 1)
62  throw InvalidAnimationConfigException(
63  "Expected " + std::to_string(DEAD + 1) + " animations but found " + std::to_string(this->animations.size())
64  );
65  if (this->delays.size() != DEAD + 1)
66  throw InvalidAnimationConfigException(
67  "Expected " + std::to_string(DEAD + 1) + " delays but found " + std::to_string(this->delays.size())
68  );
69  }

165  void Entity::update()
166  {
167  if (!this->_delay) {
168  this->_delay = this->_configs.delays[this->_animation];
169  switch (this->_animation) {
170  case DEAD:
171  case HIT:
172  if (this->_animationState + 2 == this->_configs.animations[this->_animation])
173  this->_delay = -1;
174  break;
175  default:
176  break;
177  }
178  this->_animationState = (this->_animationState + 1) % this->_configs.animations[this->_animation];
179  } else
180  this->_delay--;
181  }
