71  void Entity::move(double angle, bool dontmoveangle)
72  {
73  this->_old_position = this->_pos;
74  this->_pos.x += std::cos(angle) * this->_speed;
75  this->_pos.y += std::sin(angle) * this->_speed;
76  if (dontmoveangle)
77  return;
78  while (angle < 0)
79  angle += 2 * M_PI;
80  angle = std::fmod(angle, 2 * M_PI);
81  if (std::abs(std::cos(angle)) * this->_speed > M_PI_4)
82  this->_entity.setDirection((angle > M_PI_2 && angle < 3 * M_PI_2) ? Rendering::WEST : Rendering::EAST);
83 
84  if (angle < M_PI_4)
85  this->_pos.r = Orientation::EAST;
86  else if (angle < M_PI_4 * 3)
87  this->_pos.r = Orientation::SOUTH;
88  else if (angle < M_PI + M_PI_4)
89  this->_pos.r = Orientation::WEST;
90  else
