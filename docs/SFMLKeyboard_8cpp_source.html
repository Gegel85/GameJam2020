132  double SFMLKeyboard::getDirectionAngle() const
133  {
134  bool dirs[4] = {
135  this->actionPressed(UP),
136  this->actionPressed(DOWN),
137  this->actionPressed(LEFT),
138  this->actionPressed(RIGHT),
139  };
140 
141  reinterpret_cast<short *>(dirs)[0] *= dirs[0] != dirs[1];
142  reinterpret_cast<short *>(dirs)[1] *= dirs[2] != dirs[3];
143 
144  if (*reinterpret_cast<unsigned *>(dirs) == 0)
145  return 0;
146 
147  if (dirs[0] + dirs[1] + dirs[2] + dirs[3] == 1)
148  return (dirs[0] * -M_PI_2 + dirs[1] * M_PI_2 + dirs[2] * M_PI);
149  return (M_PI_2 + dirs[2] * M_PI_4 - dirs[3] * M_PI_4) * (dirs[1] - dirs[0]);
150  }

198  void SFMLKeyboard::_updateState()
199  {
200  if (this->_window && !this->_window->hasFocus())
201  return;
202  for (unsigned i = 0; i < NB_OF_ACTION; i++) {
203  bool press = this->actionPressed(static_cast<Action>(i));
204 
205  if (press != this->_state[i]) {
206  this->_state[i] = press;
207  this->_events.emplace_back(
208  press ? Event::EVENT_TRIGGERED : Event::EVENT_STOPPED_TRIGGER,
209  static_cast<Action>(i)
210  );
211  }
212  }
213  }
